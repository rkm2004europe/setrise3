import 'package:flutter/material.dart';
import 'package:flutter/services.dart';
import 'package:setrise/features/setrize/models/post_model.dart';
import 'package:setrise/features/setrize/models/story_model.dart';
import 'package:setrise/features/setrize/screens/widgets/post_card.dart';
import 'package:setrise/features/setrize/screens/widgets/stories_bar.dart';
import 'package:setrise/features/setrize/screens/widgets/top_bar.dart';

class SetrizeScreen extends StatefulWidget {
  const SetrizeScreen({super.key});

  @override
  State<SetrizeScreen> createState() => _SetrizeScreenState();
}

class _SetrizeScreenState extends State<SetrizeScreen> with SingleTickerProviderStateMixin {
  late List<PostModel>  _posts;
  late List<StoryModel> _stories;
  int  _idx        = 0;
  bool _showTopBar = true;
  late AnimationController _fadeCtrl;
  late Animation<double>   _fadeAnim;

  @override
  void initState() {
    super.initState();
    _posts   = PostModel.getMockPosts();
    _stories = StoryModel.getMockStories();
    _fadeCtrl = AnimationController(vsync: this,
      duration: const Duration(milliseconds: 280), value: 1.0);
    _fadeAnim = CurvedAnimation(parent: _fadeCtrl, curve: Curves.easeInOut);
  }

  @override
  void dispose() { _fadeCtrl.dispose(); super.dispose(); }

  Future<void> _next() async {
    if (_idx >= _posts.length - 1) {
      setState(() => _posts.addAll(PostModel.getMockPosts()
        .map((p) => p.copyWith(id: '${p.id}_x$_idx'))));
    }
    await _fadeCtrl.reverse();
    if (!mounted) return;
    setState(() => _idx++);
    _fadeCtrl.forward();
    HapticFeedback.selectionClick();
  }

  Future<void> _prev() async {
    if (_idx <= 0) return;
    await _fadeCtrl.reverse();
    if (!mounted) return;
    setState(() => _idx--);
    _fadeCtrl.forward();
    HapticFeedback.selectionClick();
  }

  @override
  Widget build(BuildContext context) {
    final post = _posts[_idx];

    return Scaffold(
      backgroundColor: Colors.black,
      extendBodyBehindAppBar: true,
      body: Stack(fit: StackFit.expand, children: [

        // ── PostCard ────────────────────────────────────────
        FadeTransition(opacity: _fadeAnim,
          child: PostCard(
            key: ValueKey(post.id),
            post: post,
            onUpdate: (p) => setState(() => _posts[_idx] = p),
            onSwipeNext:  _next,
            onSwipeRight: _next,
            onSwipeLeft:  _next,
            onSwipeStart: () => setState(() => _showTopBar = false),
            onSwipeEnd:   () => setState(() => _showTopBar = true),
          )),

        // ── Vertical swipe ──────────────────────────────────
        Positioned.fill(child: GestureDetector(
          behavior: HitTestBehavior.translucent,
          onVerticalDragEnd: (d) {
            final v = d.primaryVelocity ?? 0;
            if (v < -400) _next();
            if (v > 400)  _prev();
          })),

        // ── TopBar ──────────────────────────────────────────
        AnimatedSlide(
          offset: _showTopBar ? Offset.zero : const Offset(0, -1),
          duration: const Duration(milliseconds: 300),
          curve: Curves.easeInOut,
          child: AnimatedOpacity(
            opacity: _showTopBar ? 1.0 : 0.0,
            duration: const Duration(milliseconds: 200),
            child: SetrizeTopBar(
              onSearch:       () { /* TODO: search */ },
              onNotification: () { /* TODO: alerts */ },
            ))),

        // ── StoriesBar ──────────────────────────────────────
        AnimatedSlide(
          offset: _showTopBar ? Offset.zero : const Offset(0, -0.3),
          duration: const Duration(milliseconds: 300),
          curve: Curves.easeInOut,
          child: AnimatedOpacity(
            opacity: _showTopBar ? 1.0 : 0.0,
            duration: const Duration(milliseconds: 200),
            child: Align(alignment: Alignment.topCenter,
              child: Padding(
                padding: EdgeInsets.only(
                  top: MediaQuery.of(context).padding.top + 56),
                child: SetrizeStoriesBar(
                  stories: _stories,
                  onStoryTap: (s) { /* TODO: story viewer */ },
                ))))),

        // ── Progress dots ───────────────────────────────────
        Positioned(bottom: MediaQuery.of(context).padding.bottom + 8,
          left: 0, right: 0,
          child: Center(child: _Dots(
            total: _posts.length.clamp(0, 8),
            current: _idx.clamp(0, 7)))),
      ]),
    );
  }
}

class _Dots extends StatelessWidget {
  final int total, current;
  const _Dots({required this.total, required this.current});

  @override
  Widget build(BuildContext context) => Row(
    mainAxisSize: MainAxisSize.min,
    children: List.generate(total, (i) {
      final active = i == current;
      return AnimatedContainer(
        duration: const Duration(milliseconds: 250),
        margin: const EdgeInsets.symmetric(horizontal: 3),
        width: active ? 20 : 6, height: 6,
        decoration: BoxDecoration(
          color: active ? Colors.white : Colors.white30,
          borderRadius: BorderRadius.circular(3)));
    }));
}
